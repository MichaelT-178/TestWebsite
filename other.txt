<script>
    fetch("fav_covers.json")
        .then(response => {
            return response.json();
        })
        .then(data => {
            const favoriteCovers = data.FavoriteCovers;
            const theRepertoire = favoriteCovers.map((cover) => 
             `<li id="songelements">${cover.Song} (${cover.Appearance})</li>`
            //  `<li id="songelements">${cover.Song} (${cover.Appearance})</li>`
            // `<div style="margin: 10px 0;">â€¢ <a href="${link}" style= "font-size: 25px;text-decoration: none;">${web[i]}</a></div>`
        ).join('');

        document.getElementById("message").innerHTML = data.comment
        document.getElementById("message").style.color = "red";
        document.getElementById("thesongs").innerHTML = theRepertoire;
    })
    .catch(err => {
        console.error("Error loading JSON file:", err);
    });
</script>
